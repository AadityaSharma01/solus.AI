 Here is a simple C++ code snippet to reverse a string:

```cpp
#include <iostream>
#include <string>
using namespace std;

void reverse(string &s) {
    int length = s.length();
    for (int i = 0; i < length / 2; ++i) {
        char temp = s[i];
        s[i] = s[length - i - 1];
        s[length - i - 1] = temp;
    }
}

int main() {
    string str;
    cout << "Enter a string: ";
    getline(cin, str);
    reverse(str);
    cout << "Reversed string: " << str << endl;
    return 0;
}
```
